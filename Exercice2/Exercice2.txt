classe Point
	x:reel
	y:reel
	
	//constructeur de la classe point
	fonction Point(x:reel, y:reel)
	debut
	this.x=x
	this.y=y
	fin
	
	//getters et setters
	fonction getX()
	debut
	retourner x
	fin
	
	fonction setX(nouvelX: reel)
	debut
	affecter nouvelX à X
	fin
	
	fonction getY()
	debut
	retourner y
	fin
	
	fonction setY(nouvelY: reel)
	debut
	affecter nouvelY à Y
	fin
	
	//Affichage des infos du point sous forme de chaine de caractères
	fonction toString()
	debut
	afficher     "Mes coordonnées sont " + str(x) +","+str(y)
	fin
	
programme main
	point1:Point
	point2:Point
debut
	//instanciation des points test
	point1(1,2);
	point2(3,1);
	
	//test des fonctions
	afficher point1.getX()
	afficher point1.getY()
	afficher point2.getX()
	afficher point2.getY()

fin
